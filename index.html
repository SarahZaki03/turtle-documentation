الأجزاء الأساسية في بدء الرسم

import turtle

pencil = turtle.Turtle()

نبدأ بالرسم هنا

turtle.done()

ما هي الأدوات المستخدمة في الرسم؟

pencil.forward(distance) | pencil.fd(distance)

pencil.backward(distance) | pencil.bk(distance) | pencil.back(distance)

pencil.left(degree) | pencil.lt(degree)

pencil.right(degree) | pencil.rt(degree)

pencil.circle(radius, extent, steps)

pencil.goto(x, y) | pencil.setpos(x, y) | pencil.setposition(x, y)

pencil.teleport(x, y)

pencil.setx(x)

pencil.sety(y)

pencil.seth(to_angle) | pencil.setheading(to_angle)

pencil.home()

pencil.dot(size, color)

pencil.undo()

pencil.speed(speed)

ما هي الأدوات التي تخبرك بمكان وحالة القلم؟

pencil.position() | pencil.pos()

pencil.towards(x, y)

pencil.xcor()

pencil.ycor()

pencil.heading()

pencil.distance(x, y) | pencil.distance(another_pen)

ما هي الأدوات التي تتحكم بحالة القلم من خلالها؟

pencil.pendown() | pencil.pd() | pencil.down()

pencil.penup() | pencil.pu() | pencil.up()

pencil.pensize(width) | pencil.width(width)

pencil.isdown()
